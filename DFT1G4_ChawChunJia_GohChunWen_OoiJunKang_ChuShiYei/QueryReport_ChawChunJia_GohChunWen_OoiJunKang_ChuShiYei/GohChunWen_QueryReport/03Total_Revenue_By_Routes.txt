SET LINESIZE 100
SET PAGESIZE 30

COLUMN Route_ID FORMAT A10 HEADING "Route ID"
COLUMN Destination FORMAT A30 HEADING "Destination"
COLUMN Total_Revenue FORMAT 9999.99 HEADING "Total Revenue (RM)"

TTITLE LEFT 'Total Revenue Generated by Routes' -
       RIGHT 'Page No: ' FORMAT 999 SQL.PNO SKIP 2

SELECT R.Route_ID, R.Destination, SUM(T.Price) AS Total_Revenue
FROM Route R
JOIN Flight F ON R.Aircraft_ID = F.Aircraft_ID
JOIN Ticket T ON F.Passenger_ID = T.Passenger_ID
GROUP BY R.Route_ID, R.Destination;

CLEAR COLUMNS
CLEAR BREAKS
TTITLE OFF
